|Étudiant (App Mobile)|
start
:Authentification;
if (Identifiants valides ?) then (Oui)
  :Accès au Tableau de Bord;
  :Lancement du Quiz;
  
  fork
    :Passation des questions;
    :Calcul du score;
  fork again
    partition "Surveillance Active (Background)" {
      while (Test en cours ?) is (Oui)
        :Capture photos (Frontale/Arrière);
        :Enregistrement audio continu;
        :Vérification Foreground (App focus);
        :Envoi des flux vers FastAPI/Supabase;
      endwhile (Fin du test)
    }
  end fork

  :Enregistrement Géolocalisation (GPS);
  :Soumission finale des données;
else (Non)
  :Affichage Erreur;
  stop
endif

  :Analyse IA (YOLO - Détection fraude);

  :Visualisation des preuves (Photos/Audio);
  :Validation de l'intégrité;
